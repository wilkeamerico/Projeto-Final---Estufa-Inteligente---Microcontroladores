// =========================================================================
//  HANDLERS DO SERVIDOR WEB
// =========================================================================
void handleRoot() { server.send(200, "text/html", buildPage()); }

void handleSetLimites() {
  if (server.hasArg("tmax")) tempLimite = server.arg("tmax").toFloat();
  if (server.hasArg("umin")) umidadeMin = server.arg("umin").toFloat();
  if (server.hasArg("lmin")) luminMin   = server.arg("lmin").toInt();
  if (server.hasArg("gmax")) gasLimite  = server.arg("gmax").toInt();
  server.sendHeader("Location", "/");
  server.send(303);
}

void handleSimulaModo() {
  if (server.hasArg("tipo")) simulaManual = (server.arg("tipo") == "manual");
  server.sendHeader("Location", "/");
  server.send(303);
}

void handleSetSimulado() {
  if (server.hasArg("tsim")) tempSimulada = server.arg("tsim").toFloat();
  if (server.hasArg("usim")) umidSimulada = server.arg("usim").toFloat();
  server.sendHeader("Location", "/");
  server.send(303);
}

void handleBomba() {
  if(!alarmeCriticoTrava) { // Só aceita comando se não houver trava crítica
    String e = server.arg("estado");
    if      (e == "on")  { bombaManual = true; bombaModoWeb = true; }
    else if (e == "off") { bombaManual = true; bombaModoWeb = false;}
    else                 { bombaManual = false; }
  }
  server.sendHeader("Location", "/");
  server.send(303);
}

void handleCooler() {
  // Só aceita comando se não houver trava crítica e nem limite de temperatura ativo
  if(!alarmeCriticoTrava && temperatura <= tempLimite) { 
    String e = server.arg("estado");
    if      (e == "on")  { coolerManual = true; coolerModoWeb = true; }
    else if (e == "off") { coolerManual = true; coolerModoWeb = false;}
    else                 { coolerManual = false; }
  }
  server.sendHeader("Location", "/");
  server.send(303);
}

void handleLampada() {
  String e = server.arg("estado");
  if      (e == "on")  { lampadaManual = true; lampadaModoWeb = true; }
  else if (e == "off") { lampadaManual = true; lampadaModoWeb = false;}
  else                 { lampadaManual = false; }
  server.sendHeader("Location", "/");
  server.send(303);
}

void handleResetAlarme() {
  // O botão via Web desarma a trava se os valores atuais voltarem à normalidade
  if (temperatura <= 50.0 && umidade >= 30.0) {
    alarmeCriticoTrava = false;
    noTone(buzzer);
  }
  server.sendHeader("Location", "/");
  server.send(303);
}

// =========================================================================
//  SETUP
// =========================================================================
void setup() {
  Serial.begin(115200);
  delay(1000);

  Wire.begin();
  if (!display.begin(SSD1306_SWITCHCAPVCC, OLED_ADDRESS)) {
    Serial.println("Falha ao iniciar OLED SSD1306!");
    while (1);
  }
  display.clearDisplay();
  display.setTextColor(SSD1306_WHITE);
  display.setTextSize(1);
  display.setCursor(0, 0);
  display.println("Estufa Inteligente");
  display.println("Iniciando AP...");
  display.display();

  dht.begin();

  pinMode(RELE_BOMBA,   OUTPUT); digitalWrite(RELE_BOMBA,   HIGH);
  pinMode(RELE_COOLER,  OUTPUT); digitalWrite(RELE_COOLER,  HIGH);
  pinMode(RELE_LAMPADA, OUTPUT); digitalWrite(RELE_LAMPADA, HIGH);
  pinMode(buzzer, OUTPUT);

  WiFi.mode(WIFI_AP);
  WiFi.softAPConfig(ap_ip, ap_ip, IPAddress(255, 255, 255, 0));
  WiFi.softAP(ap_ssid, ap_password);

  server.on("/",            handleRoot);
  server.on("/setlimites",  handleSetLimites);
  server.on("/simulaModo",  handleSimulaModo);
  server.on("/setSimulado", handleSetSimulado);
  server.on("/bomba",       handleBomba);
  server.on("/cooler",      handleCooler);
  server.on("/lampada",     handleLampada);
  server.on("/resetAlarme", handleResetAlarme);
  server.begin();
}

// =========================================================================
//  LOOP
// =========================================================================
void loop() {
  server.handleClient();
  unsigned long atualMillis = millis();

  // --- SIMULAÇÃO DE SENSORES ---
  if (simulaManual) {
    temperatura = tempSimulada;
    umidade     = umidSimulada;
  } else {
    static float dirTemp = 0.5;
    tempSimulada += dirTemp;
    if (tempSimulada > 55.0 || tempSimulada < 20.0) dirTemp = -dirTemp;
    static float dirUmid = -1.0;
    umidSimulada += dirUmid;
    if (umidSimulada > 80.0 || umidSimulada < 25.0) dirUmid = -dirUmid;
    
    temperatura = tempSimulada;
    umidade     = umidSimulada;
  }

  luminosidade = analogRead(LDR_PIN);
  nivelGas     = analogRead(MQ_PIN);

  // --- LÓGICA DE ALARMES E PRIORIDADES ATUADORES ---
  
  // 1. Checa condição para Trava Crítica (T>50 e U<30)
  if (temperatura > 50.0 && umidade < 30.0) {
    alarmeCriticoTrava = true;
  }

  // 2. Executa Ações Baseadas na Hierarquia
  if (alarmeCriticoTrava) {
    buzzerStatus = 3;
    tone(buzzer, 1000); // Toca contínuo
    digitalWrite(RELE_COOLER, LOW); // FORÇA LIGAR (Sobrepõe manual web)
    digitalWrite(RELE_BOMBA, LOW);  // FORÇA LIGAR (Sobrepõe manual web)
  } 
  else if (temperatura > tempLimite) {
    buzzerStatus = 1;
    // Força Cooler LIGADO ignorando ajuste manual
    digitalWrite(RELE_COOLER, LOW); 
    
    // Buzzer bip curto 5s
    if (atualMillis - ultimoBip5s >= 5000) {
      ultimoBip5s = atualMillis;
      tone(buzzer, 1200, 200); 
    }
    
    // Bomba continua operando livre (obedece o web manual ou auto)
    if (bombaManual) digitalWrite(RELE_BOMBA, bombaModoWeb ? LOW : HIGH);
    else digitalWrite(RELE_BOMBA, (umidade < umidadeMin) ? LOW : HIGH);
  } 
  else if (umidade < umidadeMin) {
    buzzerStatus = 2;
    
    // Buzzer bip duplo a cada 10s
    if (estagioBipDuplo == 0 && atualMillis - ultimoBip10s >= 10000) {
      ultimoBip10s = atualMillis;
      tone(buzzer, 1000, 150);
      estagioBipDuplo = 1;
    } else if (estagioBipDuplo == 1 && atualMillis - ultimoBip10s >= 250) { 
      tone(buzzer, 1000, 150);
      estagioBipDuplo = 0;
    }

    // Tanto Bomba quanto Cooler operam livres via Web Manual ou Auto
    if (coolerManual) digitalWrite(RELE_COOLER, coolerModoWeb ? LOW : HIGH);
    else digitalWrite(RELE_COOLER, HIGH);

    if (bombaManual) digitalWrite(RELE_BOMBA, bombaModoWeb ? LOW : HIGH);
    else digitalWrite(RELE_BOMBA, LOW); // Auto padrão: liga porque umidade baixou
  } 
  else {
    // 3. Sistema Operando Normalmente Sem Alarmes
    buzzerStatus = 0;
    noTone(buzzer);
    estagioBipDuplo = 0;
    
    if (coolerManual) digitalWrite(RELE_COOLER, coolerModoWeb ? LOW : HIGH);
    else digitalWrite(RELE_COOLER, HIGH);

    if (bombaManual) digitalWrite(RELE_BOMBA, bombaModoWeb ? LOW : HIGH);
    else digitalWrite(RELE_BOMBA, HIGH);
  }

  // Controle Luz (não sofre interferência dos alarmes termo/hídricos)
  if (lampadaManual) {
    digitalWrite(RELE_LAMPADA, lampadaModoWeb ? LOW : HIGH);
  } else {
    digitalWrite(RELE_LAMPADA, (luminosidade < luminMin) ? LOW : HIGH);
  }

  // --- LEITURA FINAL DOS PINOS PARA DISPLAY/SERIAL ---
  bool bombaLig  = (digitalRead(RELE_BOMBA)   == LOW);
  bool coolerLig = (digitalRead(RELE_COOLER)  == LOW);
  bool lampLig   = (digitalRead(RELE_LAMPADA) == LOW);

  // --- EXIBIÇÃO NO OLED SSD1306 ---
  display.clearDisplay();
  display.setTextSize(1);
  
  display.setCursor(0, 0);
  display.print("T:"); display.print(temperatura,0); display.print("C (L:"); display.print(tempLimite,0); display.println(")");
  display.print("U:"); display.print(umidade,0); display.print("% (L:"); display.print(umidadeMin,0); display.println(")");
  
  display.setCursor(0, 18);
  display.print("Gas: "); display.print(nivelGas);
  display.setCursor(64, 18);
  display.print("Lz:"); display.print(lampLig ? "ON" : "OFF");

  display.setCursor(0, 32);
  display.print("Col:"); display.print(coolerLig ? "[ON]" : "[OFF]");
  display.setCursor(64, 32);
  display.print("Bmb:"); display.print(bombaLig ? "[ON]" : "[OFF]");

  display.setCursor(0, 48);
  if (alarmeCriticoTrava) display.print("ALARM: CRITICO!!");
  else if (buzzerStatus == 1) display.print("ALARM: TEMP ALTA");
  else if (buzzerStatus == 2) display.print("ALARM: UMID BAIXA");
  else display.print("SISTEMA OK");

  display.display();

  // --- ENVIO PARA O SERIAL MONITOR ---
  Serial.println("\n=== MONITORAMENTO ESTUFA ===");
  Serial.printf("Temperatura: %.1f C  | Limite: %.0f C\n", temperatura, tempLimite);
  Serial.printf("Umidade:     %.1f %% | Minimo: %.0f %%\n", umidade, umidadeMin);
  Serial.printf("CO2 ADC:     %d\n", nivelGas);
  Serial.printf("Luz status:  %s\n", lampLig ? "LIGADA" : "DESLIGADA");
  Serial.printf("Cooler:      %s\n", coolerLig ? "LIGADO" : "DESLIGADO");
  Serial.printf("Bomba:       %s\n", bombaLig ? "LIGADA" : "DESLIGADA");
  
  if (alarmeCriticoTrava) Serial.println("-> ALARME CRÍTICO ATIVADO! T>50 e U<30 (TRAVADO)");
  else if (buzzerStatus==1) Serial.println("-> ALERTA: Temp Alta (Bip 5s / Cooler Forcado)");
  else if (buzzerStatus==2) Serial.println("-> ALERTA: Umid Baixa (Bip 10s / Bomba Livre)");
  else Serial.println("-> Buzzer: Silencioso");
  Serial.println("============================");

  delay(1000);
}
